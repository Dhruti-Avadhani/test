def market_simulator(stocks, duration_seconds):
    """Background thread function to fluctuate market prices."""
    start_time = time.time()
    while time.time() - start_time < duration_seconds:
        for stock in stocks:
            stock.update_price()
        time.sleep(0.5)  # Market ticks every half second
